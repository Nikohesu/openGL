Paso 2 de 6

⌨️ Función crearEstudiante()

Función que pide datos al usuario y retorna un struct completo.

// Función para crear un estudiante desde teclado
Estudiante crearEstudiante() {
    Estudiante nuevo;

    cout << "Nombre: ";
    getline(cin, nuevo.nombre);

    cout << "Código: ";
    cin >> nuevo.codigo;

    cout << "Promedio: ";
    cin >> nuevo.promedio;

    cin.ignore();  // Limpiar buffer

    return nuevo;
}

¿Qué hace esta función?

  1. Crea un struct vacío
  2. Pide el nombre (con getline)
  3. Pide el código (con cin)
  4. Pide el promedio (con cin)
  5. Limpia el buffer
  6. Retorna el struct lleno

💡 cin.ignore(): Necesario después de cin >> si luego usaremos getline().

Paso 2 de 6

🔍 Función buscarPorCodigo()

Busca un estudiante en el vector y retorna su posición (o -1 si no existe).

// Busca por código, retorna índice o -1
int buscarPorCodigo(vector<Estudiante> &lista, int codigo) {
    for (int i = 0; i < lista.size(); i++) {
        if (lista[i].codigo == codigo) {
            return i;  // Encontrado!
        }
    }
    return -1;  // No encontrado
}

Uso:

int pos = buscarPorCodigo(lista, 1234567);
if (pos != -1) {
    mostrarEstudiante(lista[pos]);
} else {
    cout << "No encontrado";
}

¿Por qué retorna int?

0, 1, 2... Posición donde está
-1 No se encontró

⚠️ Nota: Usamos &lista (referencia) para no copiar todo el vector.

✅ Paso 2 completado: Ya tenemos las 3 funciones auxiliares.

Paso 5 de 6

📂 Función cargarDesdeArchivo()

Lee el archivo CSV y llena el vector.

// Carga estudiantes desde archivo CSV
void cargarDesdeArchivo(vector<Estudiante> &lista, string archivo) {
    ifstream f(archivo);
    string linea;

    if (f.is_open()) {
        while (getline(f, linea)) {
            Estudiante est;
            
            // Encontrar posiciones de las comas
            int pos1 = linea.find(',');
            int pos2 = linea.find(',', pos1 + 1);

            // Extraer campos
            est.nombre = linea.substr(0, pos1);
            est.codigo = stoi(linea.substr(pos1 + 1, pos2 - pos1 - 1));
            est.promedio = stof(linea.substr(pos2 + 1));

            lista.push_back(est);
        }
        f.close();
        cout << "Cargados " << lista.size() << " estudiantes." << endl;
    }
    // Si no existe el archivo, simplemente no carga nada (lista vacía)
}

#include <iostream>
#include <vector>
#include <fstream>
#include <string>
using namespace std;

struct Estudiante { string nombre; int codigo; float promedio; };

Estudiante crearEstudiante() {
    Estudiante nuevo;
    cout << "Nombre: "; getline(cin, nuevo.nombre);
    cout << "Código: "; cin >> nuevo.codigo;
    cout << "Promedio: "; cin >> nuevo.promedio; cin.ignore();
    return nuevo;
}

void mostrarEstudiante(Estudiante est) {
    cout << est.nombre << " (Código: " << est.codigo << ", Promedio: " << est.promedio << ")" << endl;
}

int buscarPorCodigo(vector<Estudiante> &lista, int codigo) {
    for (int i = 0; i < lista.size(); i++) if (lista[i].codigo == codigo) return i;
    return -1;
}

void guardarEnArchivo(vector<Estudiante> &lista, string archivo) {
    ofstream f(archivo);
    if (f.is_open()) {
        for (int i = 0; i < lista.size(); i++)
            f << lista[i].nombre << "," << lista[i].codigo << "," << lista[i].promedio << endl;
        f.close(); cout << "Guardados " << lista.size() << " estudiantes." << endl;
    }
}

void cargarDesdeArchivo(vector<Estudiante> &lista, string archivo) {
    ifstream f(archivo); string linea;
    if (f.is_open()) {
        while (getline(f, linea)) {
            Estudiante est;
            int p1 = linea.find(','), p2 = linea.find(',', p1+1);
            est.nombre = linea.substr(0, p1);
            est.codigo = stoi(linea.substr(p1+1, p2-p1-1));
            est.promedio = stof(linea.substr(p2+1));
            lista.push_back(est);
        }
        f.close(); cout << "Cargados " << lista.size() << " estudiantes." << endl;
    }
}

int main() {
    vector<Estudiante> estudiantes;
    string archivo = "estudiantes.csv";
    cargarDesdeArchivo(estudiantes, archivo);
    int opcion;
    do {
        cout << "\n=== SISTEMA DE ESTUDIANTES ===\n1. Agregar\n2. Listar (" << estudiantes.size() << ")\n3. Buscar\n4. Guardar y salir\nOpción: ";
        cin >> opcion; cin.ignore();
        switch (opcion) {
            case 1: { estudiantes.push_back(crearEstudiante()); cout << "Agregado!" << endl; break; }
            case 2: if (estudiantes.empty()) cout << "Vacío." << endl;
                    else for (int i=0; i<estudiantes.size(); i++) { cout << i+1 << ". "; mostrarEstudiante(estudiantes[i]); }
                    break;
            case 3: { int cod; cout << "Código: "; cin >> cod; int pos = buscarPorCodigo(estudiantes, cod);
                    if (pos != -1) mostrarEstudiante(estudiantes[pos]); else cout << "No encontrado." << endl; break; }
            case 4: guardarEnArchivo(estudiantes, archivo); cout << "¡Hasta luego!" << endl; break;
        }
    } while (opcion != 4);
    return 0;
}
